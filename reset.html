function requestCode(e) { e.preventDefault(); let data = null; try { data = JSON.parse(localStorage.getItem('beekeeper') || 'null'); } catch (err) {} const email = document.getElementById('reset_email')?.value || ''; const msg = document.getElementById('reset_msg'); if (!data || data.email !== email) { if (msg) { msg.textContent = 'E-post ikke funnet. Registrer deg først.'; msg.classList.remove('hidden'); } return; } const code = String(Math.floor(100000 + Math.random() * 900000)); try { localStorage.setItem('reset_code', code); localStorage.setItem('reset_email', email); } catch (err) {} if (msg) { msg.textContent = 'Kode er sendt. Skriv inn koden for å fortsette.'; msg.classList.remove('hidden'); } document.getElementById('step_request').classList.add('hidden'); document.getElementById('step_verify').classList.remove('hidden'); } function verifyCode(e) { e.preventDefault(); const codeInput = document.getElementById('reset_code')?.value || ''; const code = localStorage.getItem('reset_code') || ''; if (code && codeInput === code) { document.getElementById('step_verify').classList.add('hidden'); document.getElementById('step_setpin').classList.remove('hidden'); } } function setNewPin(e) { e.preventDefault(); const pin = document.getElementById('new_pin')?.value || ''; let data = null; try { data = JSON.parse(localStorage.getItem('beekeeper') || 'null'); } catch (err) {} const email = localStorage.getItem('reset_email') || ''; if (!data || data.email !== email) return; data.pin = pin; try { localStorage.setItem('beekeeper', JSON.stringify(data)); } catch (err) {} localStorage.removeItem('reset_code'); localStorage.removeItem('reset_email'); window.location.href = 'login.html'; }
